    <div id="notification" class="notification"></div>

    <script>
        // DOM 로드 완료 후 실행
        document.addEventListener('DOMContentLoaded', function() {
            initializeCountdown();
        });

        // 커스텀 알림 함수
        function showNotification(message, type = 'success') {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.className = `notification ${type} show`;
            
            // 3초 후 자동으로 숨기기
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }

        // 카운트다운 초기화 및 실행
        function initializeCountdown() {
            const countDownDate = new Date("2026-01-01T00:00:00").getTime();
            
            function updateCountdown() {
                try {
                    const now = new Date().getTime();
                    const distance = countDownDate - now;

                    if (distance < 0) {
                        // 카운트다운 종료
                        document.getElementById("countdownTitle").innerHTML = "D-DAY 도달 !";
                        document.getElementById("days").innerHTML = "0";
                        document.getElementById("hours").innerHTML = "0";
                        document.getElementById("minutes").innerHTML = "0";
                        document.getElementById("seconds").innerHTML = "0";
                        return;
                    }

                    // 시간 계산
                    const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                    const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                    const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                    const seconds = Math.floor((distance % (1000 * 60)) / 1000);

                    // DOM 업데이트
                    const daysEl = document.getElementById("days");
                    const hoursEl = document.getElementById("hours");
                    const minutesEl = document.getElementById("minutes");
                    const secondsEl = document.getElementById("seconds");

                    if (daysEl) daysEl.innerHTML = days;
                    if (hoursEl) hoursEl.innerHTML = hours;
                    if (minutesEl) minutesEl.innerHTML = minutes;
                    if (secondsEl) secondsEl.innerHTML = seconds;
                } catch (error) {
                    console.error('카운트다운 업데이트 오류:', error);
                }
            }

            // 초기 실행
            updateCountdown();
            
            // 1초마다 업데이트
            setInterval(updateCountdown, 1000);
        }

        // 폼 제출 함수
        function submitForm() {
            const form = document.getElementById('memberForm');
            const fields = ['myRegion', 'myChapter', 'myName', 'myContact', 'newMemberName', 'newMemberChapter', 'memberType'];
            
            // 필수 필드 검증
            let valid = true;
            let firstInvalidField = null;
            
            for (let i = 0; i < fields.length; i++) {
                const field = form.querySelector('[name=' + fields[i] + ']');
                if (!field || !field.value.trim()) {
                    valid = false;
                    if (field) {
                        field.style.borderColor = '#e53e3e';
                        if (!firstInvalidField) {
                            firstInvalidField = field;
                        }
                    }
                } else if (field) {
                    field.style.borderColor = '#e2e8f0';
                }
            }
            
            if (!valid) {
                showNotification('모든 필수 항목을 입력해주세요.', 'error');
                if (firstInvalidField) {
                    firstInvalidField.focus();
                }
                return false;
            }

            // 데이터 수집
            const data = {
                myRegion: form.querySelector('[name=myRegion]').value,
                myChapter: form.querySelector('[name=myChapter]').value,
                myName: form.querySelector('[name=myName]').value,
                myContact: form.querySelector('[name=myContact]').value,
                newMemberName: form.querySelector('[name=newMemberName]').value,
                newMemberChapter: form.querySelector('[name=newMemberChapter]').value,
                memberType: form.querySelector('[name=memberType]').value
            };

            // GET 방식으로 데이터 전송
            const params = new URLSearchParams(data);
            const url = 'https://script.google.com/macros/s/AKfycbw2p3iZJlL5CT_QlZrxlaWl-GKXB4RT79pcN67ROtBh327v8sNZnGY-fnmjjnX-EYn5/exec?' + params.toString();
            
            // 숨겨진 iframe으로 전송
            const iframe = document.createElement('iframe');
            iframe.style.display = 'none';
            iframe.src = url;
            document.body.appendChild(iframe);
            
            // 성공 알림 (변경된 메시지)
            showNotification('대표님의 정보가 성공적으로 기록되었습니다! 🎉', 'success');
            form.reset();
            
            // iframe 제거
            setTimeout(function() {
                if (document.body.contains(iframe)) {
                    document.body.removeChild(iframe);
                }
            }, 1000);
            
            return false;
        }

        // 오류 방지를 위한 추가 초기화
        window.addEventListener('load', function() {
            if (typeof initializeCountdown === 'function') {
                initializeCountdown();
            }
        });
    </script>
