    <script>
        // 새 배포된 구글 앱스 스크립트 URL (상금 재계산 버튼 제거 버전)
        const API_URL = 'https://script.google.com/macros/s/AKfycbyXXPPtJ-5T98hc63xxfLXtKqfvTK3YaUkxTyNjEbNofiWyAGf7N2GT6i4k1X2xDKUXEw/exec?action=getData';

        // 상금 계산은 구글 시트 I열 데이터를 그대로 사용

        async function loadData() {
            // 이미 업데이트 중이면 중복 실행 방지
            if (isUpdating) {
                console.log('이미 업데이트 중 - 스킵');
                return;
            }
            
            isUpdating = true;
            try {
                updateTimestamp();
                showLoading();

                // Google Apps Script는 GET 요청만 허용하고 간단한 fetch 사용
                const response = await fetch(API_URL);
                
                if (!response.ok) {
                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);
                }
                
                // Google Apps Script는 JSON이 아닌 텍스트로 응답할 수 있음
                const text = await response.text();
                
                let jsonData;
                try {
                    // HTML이 포함된 응답인 경우 처리
                    if (text.includes('<!DOCTYPE html>') || text.includes('<html>')) {
                        throw new Error('Google Apps Script 웹앱 설정 확인 필요');
                    }
                    jsonData = JSON.parse(text);
                } catch (parseError) {
                    throw new Error('응답 파싱 실패: ' + text.substring(0, 100));
                }
                
                // Google Apps Script 오류 응답 확인
                if (jsonData.error) {
                    throw new Error('서버 오류: ' + jsonData.error);
                }
                
                // 실시간 동적 연동을 위한 응답 구조 확인
                if (jsonData.success && jsonData.data) {
                    // 새로운 응답 구조 (실시간 연동)
                    jsonData = jsonData.data;
                }
                
                // 데이터 형식 검증 및 처리
                if (!Array.isArray(jsonData)) {
                    console.error('받은 데이터 형식:', typeof jsonData);
                    console.error('받은 데이터 내용:', jsonData);
                    
                    // 빈 객체나 null인 경우 빈 배열로 처리
                    if (jsonData === null || jsonData === undefined || (typeof jsonData === 'object' && Object.keys(jsonData).length === 0)) {
                        console.log('빈 데이터 - 빈 배열로 처리');
                        jsonData = [];
                    } else {
                        throw new Error('데이터가 배열 형식이 아닙니다. 받은 타입: ' + typeof jsonData);
                    }
                }
                
                // 빈 배열인 경우 처리
                if (jsonData.length === 0) {
                    console.log('구글 시트에서 빈 데이터를 받았습니다');
                    allData = [];
                    updateAllStats();
                    updateKPIs();
                    updateCharts();
                    updateFilterOptions();
                    return;
                }
                
                // 원본 데이터의 컬럼명 확인
                console.log('원본 데이터 컬럼명:', Object.keys(jsonData[0]));
                console.log('첫 번째 행 데이터:', jsonData[0]);
                console.log('받은 데이터 개수:', jsonData.length);
                
                // 데이터 매핑 (상금 데이터 포함)
                const newData = jsonData.map(row => ({
                    timestamp: row.timestamp || '',
                    myRegion: row.myRegion || '',
                    myChapter: row.myChapter || '',
                    myName: row.myName || '',
                    myContact: row.myContact || '',
                    newMemberName: row.newMemberName || '',
                    newMemberChapter: row.newMemberChapter || '',
                    memberType: row.memberType || '',
                    rewardAmount: parseFloat(row.rewardAmount) // 구글 시트의 I열 상금 데이터 (0으로 대체하지 않음)
                }));
                
                // 데이터 변경 감지
                const dataChanged = JSON.stringify(newData) !== JSON.stringify(allData);
                
                console.log('=== 데이터 변경 감지 ===');
                console.log('데이터 변경됨:', dataChanged);
                
                allData = newData;
                
                console.log('=== 데이터 로딩 디버깅 ===');
                console.log('원본 JSON 데이터 개수:', jsonData.length);
                console.log('매핑된 allData 개수:', allData.length);
                console.log('원본 JSON 데이터 샘플:', jsonData.slice(0, 3));
                console.log('매핑된 데이터 샘플:', allData.slice(0, 3));
                console.log('지역 목록:', [...new Set(allData.map(row => row.myRegion).filter(Boolean))]);
                console.log('멤버 타입 목록:', [...new Set(allData.map(row => row.memberType).filter(Boolean))]);
                
                // 멤버 타입별 상세 분석
                const memberTypeAnalysis = {};
                allData.forEach(row => {
                    const type = row.memberType;
                    if (type && type.trim()) {
                        memberTypeAnalysis[type] = (memberTypeAnalysis[type] || 0) + 1;
                    }
                });
                console.log('멤버 타입별 분석:', memberTypeAnalysis);
                
                // 상금 데이터는 calculateReward 함수에서만 계산
                
                // 데이터가 변경되었을 때만 UI 업데이트
                if (dataChanged) {
                    console.log('데이터 변경 감지됨 - UI 업데이트 시작');
                    
                    // 이전 데이터 저장 (변화율 계산용)
                    previousData = [...allData];
                    
                    // 필터링된 데이터 업데이트
                    applyFilters();
                    
                    updateAllStats();
                    updateCharts();
                    
                    // 상금 계산은 한 번만
                    const rewardData = calculateReward(allData);
                    updateRewardDisplay(rewardData);
                    
                    console.log('UI 업데이트 완료');
                } else {
                    console.log('데이터 변경 없음 - UI 업데이트 스킵');
                }
                updateFilterOptions();
                
                // 알림 표시
                const newCount = allData.length - (previousData.length || 0);
                if (newCount > 0) {
                    showNotification(`🆕 ${newCount}명의 새로운 멤버가 추가되었습니다!`);
                }
                
                console.log('Google Apps Script 연결 성공:', allData.length, '건');
                lastUpdateTime = new Date();
                
            } catch (error) {
                console.error('Google Apps Script 연결 실패:', error.message);
                
                // 기존 데이터가 있으면 에러를 표시하지 않고 기존 데이터 유지
                if (allData.length > 0) {
                    console.log('기존 데이터 유지 - 네트워크 오류 무시');
                    return;
                }
                
                // 초기 로딩 실패 시 에러 표시
                console.error('초기 데이터 로딩 실패 - 구글 시트 연결을 확인해주세요');
                showError('구글 시트 연결 실패: ' + error.message);
            } finally {
                isUpdating = false;
                hideLoading();
            }
        }

            // 페이지 상단에 에러 메시지 표시
            const errorDiv = document.createElement('div');
            errorDiv.style.cssText = `

            errorDiv.textContent = `⚠️ ${message}`;
            document.body.appendChild(errorDiv);
            
            // 5초 후 에러 메시지 제거
            setTimeout(() => {
                if (errorDiv.parentNode) {
                    errorDiv.parentNode.removeChild(errorDiv);
                }
            }, 5000);
        }


        function updateTimestamp() {
            const now = new Date();
            const timeString = now.toLocaleString('ko-KR', {
                year: 'numeric',
                month: '2-digit',
                day: '2-digit',
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit'
            });
            document.getElementById('lastUpdateTime').textContent = `마지막 업데이트: ${timeString}`;
        }


        function updateRealtimeStatus() {
            const statusElement = document.getElementById('realtimeStatus');
            if (statusElement) {
                if (isUpdating) {
                    statusElement.innerHTML = '🔄 실시간 업데이트 중...';
                    statusElement.style.color = '#007bff';
                } else if (lastUpdateTime) {
                    const timeDiff = Math.floor((new Date() - lastUpdateTime) / 1000);
                    if (timeDiff < 60) {
                        statusElement.innerHTML = `✅ 실시간 연동 중 (${timeDiff}초 전 업데이트)`;
                        statusElement.style.color = '#28a745';
                    } else {
                        statusElement.innerHTML = `⚠️ 실시간 연동 중 (${Math.floor(timeDiff / 60)}분 전 업데이트)`;
                        statusElement.style.color = '#ffc107';
                    }
                } else {
                    statusElement.innerHTML = '❌ 실시간 연동 끊김';
                    statusElement.style.color = '#dc3545';
                }
            }
        }

        // 필터 적용 함수
        function applyFilters() {
            const regionFilter = document.getElementById('regionFilter').value;
            const chapterFilter = document.getElementById('chapterFilter').value;
            const memberTypeFilter = document.getElementById('memberTypeFilter').value;
            
            filteredData = allData.filter(row => {
                // 지역 필터
                if (regionFilter && row.myRegion !== regionFilter) return false;
                
                // 챕터 필터
                if (chapterFilter && row.myChapter !== chapterFilter) return false;
                
                // 멤버 타입 필터
                if (memberTypeFilter && !row.memberType.includes(memberTypeFilter)) return false;
                
                return true;
            });
        }

        function calculateReward(data) {
            let total = 0;
            console.log('calculateReward 호출됨, 데이터 개수:', data.length);
            data.forEach((row, index) => {
                const amount = parseFloat(row.rewardAmount);
                console.log(`${index + 1}. ${row.myName} -> ${row.newMemberName}: ${amount}원 (지역: ${row.myRegion})`);
                // 이상한 값 체크
                if (amount !== 160000 && amount !== 128000) {
                    console.warn(`⚠️ 이상한 상금 값 발견: ${amount}원 (${row.myName} -> ${row.newMemberName})`);
                }
                total += amount;
            });
            console.log('최종 총합:', total.toLocaleString(), '원');
            return { totalReward: total };
        }

        // KPI 업데이트
        function updateKPIs() {
            const data = filteredData.length > 0 ? filteredData : allData;
            
            // 총 멤버수
            const totalMembers = data.length;
            document.getElementById('totalMembers').textContent = totalMembers.toLocaleString();
            
            // 최다 가디언 지역 계산
            const regionCount = {};
            data.forEach(row => {
                const region = row.myRegion;
                if (region && region.trim()) {
                    regionCount[region] = (regionCount[region] || 0) + 1;
                }
            });
            const topRegion = Object.entries(regionCount).reduce((a, b) => regionCount[a[0]] > regionCount[b[0]] ? a : b, ['-', 0]);
            document.getElementById('topRegion').textContent = topRegion[0];
            
            // 최다 가디언 챕터 계산
            const chapterCount = {};
            data.forEach(row => {
                const chapter = row.myChapter;
                if (chapter && chapter.trim()) {
                    chapterCount[chapter] = (chapterCount[chapter] || 0) + 1;
                }
            });
            const topChapter = Object.entries(chapterCount).reduce((a, b) => chapterCount[a[0]] > chapterCount[b[0]] ? a : b, ['-', 0]);
            document.getElementById('topChapter').textContent = topChapter[0];
            
            // 전주 대비 증감 현황 계산
            const weekAgo = new Date();
            weekAgo.setDate(weekAgo.getDate() - 7);
            const currentWeekData = data.filter(row => new Date(row.timestamp) >= weekAgo);
            const previousWeekData = previousData.filter(row => {
                const rowDate = new Date(row.timestamp);
                const twoWeeksAgo = new Date();
                twoWeeksAgo.setDate(twoWeeksAgo.getDate() - 14);
                return rowDate >= twoWeeksAgo && rowDate < weekAgo;
            });
            
            const weeklyChange = currentWeekData.length - previousWeekData.length;
            const changeElement = document.getElementById('weeklyChange');
            const indicatorElement = document.getElementById('weeklyChangeIndicator');
            
            if (weeklyChange > 0) {
                changeElement.textContent = `+${weeklyChange}명`;
                indicatorElement.innerHTML = '<span>↗</span> <span>전주 대비</span>';
                indicatorElement.className = 'kpi-change positive';
            } else if (weeklyChange < 0) {
                changeElement.textContent = `${weeklyChange}명`;
                indicatorElement.innerHTML = '<span>↘</span> <span>전주 대비</span>';
                indicatorElement.className = 'kpi-change negative';
            } else {
                changeElement.textContent = '0명';
                indicatorElement.innerHTML = '<span>→</span> <span>전주 대비</span>';
                indicatorElement.className = 'kpi-change neutral';
            }
            
            // 변화율 업데이트
            updateKPIChanges(totalMembers, topRegion[1], topChapter[1], weeklyChange);
        }

        // KPI 변화율 업데이트
        function updateKPIChanges(totalMembers, topRegionCount, topChapterCount, weeklyChange) {
            // 총 가디언 변화율 (이전 데이터와 비교)
            const previousTotal = previousData.length;
            const totalChange = previousTotal > 0 ? ((totalMembers - previousTotal) / previousTotal * 100) : 0;
            
            const totalChangeElement = document.getElementById('totalMembersChange');
            if (totalChangeElement) {
                if (totalChange > 0) {
                    totalChangeElement.innerHTML = `<span>↗</span> <span>+${totalChange.toFixed(1)}%</span>`;
                    totalChangeElement.className = 'kpi-change positive';
                } else if (totalChange < 0) {
                    totalChangeElement.innerHTML = `<span>↘</span> <span>${totalChange.toFixed(1)}%</span>`;
                    totalChangeElement.className = 'kpi-change negative';
                } else {
                    totalChangeElement.innerHTML = `<span>→</span> <span>0%</span>`;
                    totalChangeElement.className = 'kpi-change neutral';
                }
            }
        }

        // 더보기 기능이 있는 바차트 생성
        function createBarChartWithShowMore(containerId, data, type) {
            const container = document.getElementById(containerId);
            if (!container) return;
            
            if (data.length === 0) {
                container.innerHTML = '<div class="error">데이터가 없습니다.</div>';
                return;
            }
            
            const maxValue = Math.max(...data.map(([, count]) => count));
            const top5Data = data.slice(0, 5);
            const remainingData = data.slice(5);
            
            let html = '';
            
            // 스크롤 가능한 바차트 컨테이너
            html += '<div class="scrollable-bar-chart">';
            
            // Top 5 표시
            top5Data.forEach(([name, count]) => {
                const percentage = (count / maxValue) * 100;
                const isMobile = window.innerWidth <= 768;
                const adjustedPercentage = isMobile && count === 1 ? Math.max(percentage, 24) : percentage;
                
                html += `
                    <div class="bar-item">
                        <div class="bar-label">${name}</div>
                        <div class="bar-track">
                            <div class="bar-fill" style="width: ${adjustedPercentage}%">
                                <div class="bar-value">${count}</div>
                            </div>
                        </div>
                        <div class="bar-count">${count}</div>
                    </div>
                `;
            });
            
            // 추가 항목들 (처음에는 숨김)
            if (remainingData.length > 0) {
                html += `
                    <div class="additional-items" id="${containerId}_additional" style="display: none;">
                `;
                
                remainingData.forEach(([name, count]) => {
                    const percentage = (count / maxValue) * 100;
                    
                    html += `
                        <div class="bar-item">
                            <div class="bar-label">${name}</div>
                            <div class="bar-track">
                                <div class="bar-fill" style="width: ${percentage}%">
                                    <div class="bar-value">${count}</div>
                                </div>
                            </div>
                            <div class="bar-count">${count}</div>
                        </div>
                    `;
                });
                
                html += `
                    </div>
                `;
            }
            
            html += '</div>'; // scrollable-bar-chart 닫기
            
            // 더보기 텍스트 (박스 맨 아래 고정)
            if (remainingData.length > 0) {
                html += `
                    <div class="show-more-container">
                        <span class="show-more-text" onclick="toggleShowMore('${containerId}', '${type}')">
                            + 더보기 (${remainingData.length}개)
                            <span class="show-more-icon">▼</span>
                        </span>
                    </div>
                `;
            }
            
            container.innerHTML = html;
        }

        // 더보기 토글 함수
        function toggleShowMore(containerId, type) {
            const additionalItems = document.getElementById(`${containerId}_additional`);
            const showMoreText = document.querySelector(`#${containerId} .show-more-text`);
            const showMoreIcon = document.querySelector(`#${containerId} .show-more-icon`);
            const scrollableChart = document.querySelector(`#${containerId} .scrollable-bar-chart`);
            
            if (additionalItems && showMoreText && showMoreIcon && scrollableChart) {
                if (additionalItems.style.display === 'none') {
                    // 더보기 클릭 시 추가 항목 표시
                    additionalItems.style.display = 'block';
                    showMoreText.innerHTML = `접기 <span class="show-more-icon">▲</span>`;
                    showMoreIcon.style.transform = 'rotate(180deg)';
                    
                    // 스크롤 높이를 늘려서 모든 항목을 볼 수 있도록 함
                    const isMobile = window.innerWidth <= 768;
                    scrollableChart.style.maxHeight = isMobile ? '250px' : '300px';
                    
                    // 스크롤을 맨 위로 이동 (모든 항목을 처음부터 볼 수 있도록)
                    setTimeout(() => {
                        scrollableChart.scrollTop = 0;
                    }, 100);
                } else {
                    // 접기 클릭 시 추가 항목 숨김
                    additionalItems.style.display = 'none';
                    showMoreText.innerHTML = `+ 더보기 (${additionalItems.children.length}개) <span class="show-more-icon">▼</span>`;
                    showMoreIcon.style.transform = 'rotate(0deg)';
                    
                    // 스크롤 높이를 원래대로 복원
                    const isMobile = window.innerWidth <= 768;
                    scrollableChart.style.maxHeight = isMobile ? '140px' : '160px';
                    
                    // 스크롤을 맨 위로 이동
                    scrollableChart.scrollTop = 0;
                }
            }
        }

        // 지역별 분포 막대차트 업데이트
        function updateRegionDistribution() {
            const data = filteredData.length > 0 ? filteredData : allData;
            const regionCount = {};
            
            console.log('지역별 분포 업데이트 - 총 데이터:', data.length, '건');
            
            data.forEach(row => {
                const region = row.myRegion;
                if (region && region.trim()) {
                    regionCount[region] = (regionCount[region] || 0) + 1;
                }
            });
            
            console.log('지역별 카운트:', regionCount);

            const total = Object.values(regionCount).reduce((sum, count) => sum + count, 0);
            const maxCount = Math.max(...Object.values(regionCount));
            
            // 실제 데이터에서 지역 목록 추출 (중복 제거 및 정렬)
            const uniqueRegions = [...new Set(Object.keys(regionCount))].sort();
            
            // 지역별 데이터 정렬 (멤버 수 기준 내림차순)
            const sortedRegions = uniqueRegions.map(region => ({
                name: region,
                count: regionCount[region] || 0,
                percentage: total > 0 ? ((regionCount[region] || 0) / total * 100).toFixed(1) : 0
            })).sort((a, b) => b.count - a.count);

            const container = document.getElementById('regionDistributionList');
            
            if (total === 0) {
                container.innerHTML = '<div class="error">데이터가 없습니다.</div>';
                return;
            }

            let html = '';
            sortedRegions.forEach(region => {
                const percentage = maxCount > 0 ? (region.count / maxCount * 100) : 0;
                let activityClass = '';
                let icon = '🏢';
                
                if (region.count > 0) {
                    if (percentage > 70) {
                        activityClass = 'high-activity';
                        icon = '🔥';
                    } else if (percentage > 30) {
                        activityClass = 'medium-activity';
                        icon = '📈';
                    } else {
                        activityClass = 'low-activity';
                        icon = '📊';
                    }
                } else {
                    activityClass = 'zero-activity';
                    icon = '📍';
                }

                html += `
                    <div class="region-distribution-item ${activityClass}" 
                         onclick="showRegionDrilldown('${region.name}')"
                         data-region="${region.name}" 
                         data-count="${region.count}">
                        <div class="region-distribution-icon">${icon}</div>
                        <div class="region-distribution-info">
                            <div class="region-distribution-name">${region.name}</div>
                            <div class="region-distribution-details">
                                <span>멤버 ${region.count}명</span>
                                <span class="region-distribution-percentage">(${region.percentage}%)</span>
                            </div>
                        </div>
                        <div class="region-distribution-bar">
                            <div class="region-distribution-bar-fill ${activityClass}" 
                                 style="width: ${percentage}%"></div>
                        </div>
                        <div class="region-distribution-count">${region.count}</div>
                    </div>
                `;
            });

            container.innerHTML = html;
        }

                console.log('첫 번째 행의 멤버 타입:', data[0].memberType);
            }
            
            data.forEach(row => {
                const memberType = row.memberType;
                if (memberType && memberType.trim()) {
                    memberTypeCount[memberType] = (memberTypeCount[memberType] || 0) + 1;
                } else {

                }
            });
            
            console.log('멤버 타입별 카운트:', memberTypeCount);

            const total = Object.values(memberTypeCount).reduce((sum, count) => sum + count, 0);
            document.getElementById('donutTotal').textContent = total;

            if (total === 0) {
                document.getElementById('memberTypeDonut').innerHTML = '<div class="error">데이터가 없습니다.</div>';
                return;
            }

            const colors = ['#d12031', '#f59e0b', '#10b981', '#3b82f6', '#8b5cf6', '#ef4444'];
            const sortedTypes = Object.entries(memberTypeCount).sort(([,a], [,b]) => b - a);
            
            console.log('정렬된 멤버 타입:', sortedTypes);
            
            let currentAngle = 0;
            let html = '';
            let legendHtml = '';

            sortedTypes.forEach(([type, count], index) => {
                console.log(`처리 중인 멤버 타입: ${type}, 개수: ${count}, 인덱스: ${index}`);
                const percentage = (count / total) * 100;
                const angle = (percentage / 100) * 360;

                const radius = 80;
                const innerRadius = 50;
                const centerX = 100;
                const centerY = 100;
                
                const startAngle = currentAngle;
                const endAngle = currentAngle + angle;
                
                const x1 = centerX + radius * Math.cos((startAngle * Math.PI) / 180);
                const y1 = centerY + radius * Math.sin((startAngle * Math.PI) / 180);
                const x2 = centerX + radius * Math.cos((endAngle * Math.PI) / 180);
                const y2 = centerY + radius * Math.sin((endAngle * Math.PI) / 180);
                
                const x3 = centerX + innerRadius * Math.cos((endAngle * Math.PI) / 180);
                const y3 = centerY + innerRadius * Math.sin((endAngle * Math.PI) / 180);
                const x4 = centerX + innerRadius * Math.cos((startAngle * Math.PI) / 180);
                const y4 = centerY + innerRadius * Math.sin((startAngle * Math.PI) / 180);
                
                const largeArcFlag = angle > 180 ? 1 : 0;
                
                const pathData = `M ${x1} ${y1} A ${radius} ${radius} 0 ${largeArcFlag} 1 ${x2} ${y2} L ${x3} ${y3} A ${innerRadius} ${innerRadius} 0 ${largeArcFlag} 0 ${x4} ${y4} Z`;
                
                html += `<path class="donut-segment" d="${pathData}" fill="${color}" data-type="${type}" data-count="${count}" onclick="showMemberTypeDrilldown('${type}')"></path>`;
                
                legendHtml += `
                    <div class="donut-legend-item">
                        <div class="donut-legend-color" style="background: ${color}"></div>
                        <div class="donut-legend-label">${type}</div>
                        <div class="donut-legend-value">${count}</div>
                    </div>
                `;
                
                console.log(`범례 항목 추가: ${type} (${count}명) - 색상: ${color}`);
                
                currentAngle += angle;
            });

            document.querySelector('#memberTypeDonut svg').innerHTML = html;
            
            const legendElement = document.getElementById('donutLegend');
            if (legendElement) {
                legendElement.innerHTML = legendHtml;
                console.log('범례 업데이트 완료');
            } else {
                console.error('범례 요소를 찾을 수 없습니다!');
            }
            
            console.log('생성된 범례 HTML:', legendHtml);
            console.log('범례 요소:', legendElement);
        }

        // 토너먼트 토글 함수
        function toggleComparisonMode() {
            const toggle = document.getElementById('comparisonToggle');
            const controls = document.getElementById('comparisonControls');
            const executeSection = document.getElementById('challengeExecuteSection');
            
            if (toggle && controls) {
                if (toggle.classList.contains('active')) {
                    // 토너먼트 모드 비활성화
                    toggle.classList.remove('active');
                    toggle.textContent = '🎮 토너먼트 시작';
                    controls.classList.remove('show');
                    executeSection.classList.remove('show');
                    
                    // 결과 숨기기
                    const results = document.getElementById('comparisonResults');
                    if (results) {
                        results.classList.remove('show');
                    }
                    
                    // 차이점 섹션 초기화
                    const diffValue = document.getElementById('comparisonDiffValue');
                    const diffLabel = document.getElementById('comparisonDiffLabel');
                    const diffContainer = document.getElementById('comparisonDifference');
                    
                    if (diffValue && diffLabel && diffContainer) {
                        diffValue.textContent = '토너먼트를 시작해볼까요?';
                        diffValue.className = 'comparison-difference-value initial-state';
                        diffLabel.textContent = '챌린지 토너먼트';
                        diffContainer.className = 'comparison-difference';
                    }
                } else {
                    // 토너먼트 모드 활성화
                    toggle.classList.add('active');
                    toggle.textContent = '🏁 토너먼트 종료';
                    controls.classList.add('show');
                    
                    // 옵션 업데이트
                    updateComparisonOptions();
                }
            }
        }

        // 비교 옵션 업데이트
        function updateComparisonOptions() {
            const type = document.getElementById('comparisonType').value;

                case 'region':

                    break;
                case 'chapter':

                    break;
                case 'memberType':
                    options = [...new Set(allData.map(row => row.memberType).filter(Boolean))];

            // 첫 번째 항목 옵션 업데이트

            const executeSection = document.getElementById('challengeExecuteSection');
            if (executeSection) {
                executeSection.classList.add('show');
            }
        }

        // 챌린지 실행
        function executeChallenge() {
            const type = document.getElementById('comparisonType').value;
            const item1 = document.getElementById('comparisonItem1').value;
            const item2 = document.getElementById('comparisonItem2').value;
            
            if (!item1 || !item2) {
                alert('두 항목을 모두 선택해주세요!');
                return;
            }
            
            if (item1 === item2) {
                alert('서로 다른 항목을 선택해주세요!');
                return;
            }
            
            // 로딩 표시
            showTournamentLoading();
            
            // 실제 비교 실행 (약간의 지연을 두어 로딩 효과)
            setTimeout(() => {
                performComparison(type, item1, item2);
                hideTournamentLoading();
            }, 1000);
        }

        // 비교 실행
        function performComparison(type, item1, item2) {
            const stats1 = calculateComparisonStats(allData, type, item1);
            const stats2 = calculateComparisonStats(allData, type, item2);
            
            // 결과 표시
            document.getElementById('comparisonTitle1').textContent = item1;
            document.getElementById('comparisonValue1').textContent = stats1.total;
            displayComparisonStats('comparisonStats1', stats1);
            
            document.getElementById('comparisonTitle2').textContent = item2;
            document.getElementById('comparisonValue2').textContent = stats2.total;
            displayComparisonStats('comparisonStats2', stats2);
            
            // 차이점 계산 및 표시
            const difference = stats1.total - stats2.total;
            const diffValue = document.getElementById('comparisonDiffValue');
            const diffLabel = document.getElementById('comparisonDiffLabel');
            const diffContainer = document.getElementById('comparisonDifference');
            
            if (difference > 0) {
                diffValue.innerHTML = `🏆 <span>${item1}</span>이(가) <span>${difference}명</span> 더 많습니다!`;
                diffContainer.className = 'comparison-difference positive';
                diffLabel.textContent = `${item1} 승리!`;
            } else if (difference < 0) {
                diffValue.innerHTML = `🏆 <span>${item2}</span>이(가) <span>${Math.abs(difference)}명</span> 더 많습니다!`;
                diffContainer.className = 'comparison-difference negative';
                diffLabel.textContent = `${item2} 승리!`;
            } else {
                diffValue.innerHTML = `🤝 <span>무승부!</span> <span>동점</span>입니다!`;
                diffContainer.className = 'comparison-difference neutral';
                diffLabel.textContent = '무승부!';
            }
            
            // 결과 섹션 표시
            const results = document.getElementById('comparisonResults');
            if (results) {
                results.classList.add('show');
            }
        }

        // 비교 통계 계산
        function calculateComparisonStats(data, type, item) {
            let filteredData = [];
            
            switch (type) {
                case 'region':
                    filteredData = data.filter(row => row.myRegion === item);
                    break;
                case 'chapter':
                    filteredData = data.filter(row => row.myChapter === item);
                    break;
                case 'memberType':
                    filteredData = data.filter(row => row.memberType === item);
                    break;
            }
            
            const regions = [...new Set(filteredData.map(row => row.myRegion).filter(Boolean))];
            const chapters = [...new Set(filteredData.map(row => row.myChapter).filter(Boolean))];
            const memberTypes = [...new Set(filteredData.map(row => row.memberType).filter(Boolean))];
            
            return {
                total: filteredData.length,
                regions: regions.length,
                chapters: chapters.length,
                memberTypes: memberTypes.length,
                avgPerRegion: regions.length > 0 ? (filteredData.length / regions.length).toFixed(1) : 0,
                avgPerChapter: chapters.length > 0 ? (filteredData.length / chapters.length).toFixed(1) : 0
            };
        }
