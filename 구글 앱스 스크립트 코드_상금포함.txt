// 구글 앱스 스크립트 코드 (상금 계산 포함)
// 이 코드를 구글 앱스 스크립트에 복사하여 사용하세요

function doGet(e) {
  try {
    // 스프레드시트 연결
    const sheet = SpreadsheetApp.getActiveSheet();
    
    // 현재 시간
    const timestamp = new Date();
    
    // 파라미터에서 데이터 추출
    const data = {
      timestamp: timestamp,
      myRegion: e.parameter.myRegion || '',
      myChapter: e.parameter.myChapter || '',
      myName: e.parameter.myName || '',
      myContact: e.parameter.myContact || '',
      newMemberName: e.parameter.newMemberName || '',
      newMemberChapter: e.parameter.newMemberChapter || '',
      memberType: e.parameter.memberType || '',
      // 상금 관련 데이터
      regionType: e.parameter.regionType || '',
      rewardAmount: e.parameter.rewardAmount || 0,
      rewardRate: e.parameter.rewardRate || '0%',
      membershipFee: e.parameter.membershipFee || 1760000
    };
    
    // 데이터를 배열로 변환하여 시트에 추가
    const rowData = [
      data.timestamp,
      data.myRegion,
      data.myChapter,
      data.myName,
      data.myContact,
      data.newMemberName,
      data.newMemberChapter,
      data.memberType,
      data.regionType,
      data.rewardAmount,
      data.rewardRate,
      data.membershipFee
    ];
    
    // 시트에 데이터 추가
    sheet.appendRow(rowData);
    
    // 성공 응답
    return ContentService
      .createTextOutput(JSON.stringify({success: true, message: '데이터가 성공적으로 저장되었습니다.'}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    // 오류 응답
    return ContentService
      .createTextOutput(JSON.stringify({success: false, message: '오류가 발생했습니다: ' + error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// 데이터 조회 함수 (대시보드용)
function getData() {
  try {
    const sheet = SpreadsheetApp.getActiveSheet();
    const data = sheet.getDataRange().getValues();
    
    // 헤더 제거하고 데이터만 반환
    const headers = data[0];
    const rows = data.slice(1);
    
    // 데이터를 객체 배열로 변환
    const result = rows.map(row => {
      const obj = {};
      headers.forEach((header, index) => {
        obj[header] = row[index];
      });
      return obj;
    });
    
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// 시트 초기화 함수 (첫 실행 시 사용)
function initializeSheet() {
  const sheet = SpreadsheetApp.getActiveSheet();
  
  // 헤더 설정
  const headers = [
    'timestamp',
    'myRegion',
    'myChapter', 
    'myName',
    'myContact',
    'newMemberName',
    'newMemberChapter',
    'memberType',
    'regionType',
    'rewardAmount',
    'rewardRate',
    'membershipFee'
  ];
  
  // 헤더가 없으면 추가
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(headers);
    
    // 헤더 스타일링
    const headerRange = sheet.getRange(1, 1, 1, headers.length);
    headerRange.setBackground('#4285f4');
    headerRange.setFontColor('white');
    headerRange.setFontWeight('bold');
  }
}

// 총 상금 계산 함수
function calculateTotalReward() {
  try {
    const sheet = SpreadsheetApp.getActiveSheet();
    const data = sheet.getDataRange().getValues();
    
    let totalReward = 0;
    let directMembers = 0;
    let regionalMembers = 0;
    
    // 헤더 제외하고 데이터 처리
    for (let i = 1; i < data.length; i++) {
      const rewardAmount = parseFloat(data[i][9]) || 0; // rewardAmount 컬럼
      const regionType = data[i][8]; // regionType 컬럼
      
      totalReward += rewardAmount;
      
      if (regionType === '직영') {
        directMembers++;
      } else if (regionType === '지역') {
        regionalMembers++;
      }
    }
    
    return {
      totalReward: totalReward,
      directMembers: directMembers,
      regionalMembers: regionalMembers,
      directReward: directMembers * 176000, // 17.6만원
      regionalReward: regionalMembers * 140800 // 14.08만원
    };
    
  } catch (error) {
    return {error: error.toString()};
  }
}

// 상금 통계 조회 함수
function getRewardStats() {
  try {
    const stats = calculateTotalReward();
    return ContentService
      .createTextOutput(JSON.stringify(stats))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({error: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
